Datalog Rules for JSFox

 

[AllocRule]

VarPointsTo(var, heap) :-

Reachable(methHeap), Alloc(var, heap, methHeap).

 

[MoveRule]

VarPointsTo(to, heap)  :-

Move(to, from), VarPointsTo(from, heap).

 

[StoreRule]

FldPointsTo(baseH, fld, heap)  :-

Store(base, fld, from), VarPointsTo(from, heap),

VarPointsTo(base, baseH).

 

[LoadRule]

VarPointsTo(to, heap)  :-

Load(to, fld, base), VarPointsTo(base, baseH),

FldPointsTo(baseH, fld, heap).

 

[FunctionCallRule]

Reachable(toMethHeap),

CallGraph(invo, toMethHeap)  :-

VCall(var, invo, inMethHeap), Reachable(inMethHeap),

VarPointsTo(var, toMethHeap).

 

[PropertyGetSetFunctionsRule]

Reachable(toMethHeap),

CallGraph(invo, toMethHeap)  :-

VCall_Prop(var, fld, invo, dir,inMethHeap), Reachable(inMethHeap),

VarPointsTo(var, baseH), ObjFldPointsTo(baseH, fld, toMethHeap).

 

[FunctionArgsAssignRule]

InterProcAssign(to, from)  :-

CallGraph(invo, methHeap),

FormalArg(methHeap, n, to),ActualArg(invo, n, from).

 

[FunctionReturnRule]

InterProcAssign(to, from)  :-

CallGraph(invo, methHeap),

FormalReturn(methHeap, from), ActualReturn(invo, to).

 

[InterProcAssignRule]

VarPointsTo(to, heap)  :-

InterProcAssign(to, from),

VarPointsTo(from, heap).
